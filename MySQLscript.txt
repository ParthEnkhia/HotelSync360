SET FOREIGN_KEY_CHECKS = 1;

-- 1) Properties (multi-hotel support)
CREATE TABLE PROPERTY (
    property_id INT PRIMARY KEY AUTO_INCREMENT,
    property_name VARCHAR(150) NOT NULL,
    address_line1 VARCHAR(200),
    city VARCHAR(100),
    state VARCHAR(100),
    country VARCHAR(100),
    status ENUM('ACTIVE', 'INACTIVE') DEFAULT 'ACTIVE',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- 2) Zone
CREATE TABLE ZONE (
    zone_id INT PRIMARY KEY AUTO_INCREMENT,
    property_id INT NOT NULL,
    zone_name VARCHAR(100) NOT NULL,
    zone_category VARCHAR(50),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    CONSTRAINT fk_zone_property
        FOREIGN KEY (property_id) REFERENCES PROPERTY(property_id)
        ON DELETE CASCADE ON UPDATE CASCADE,
    UNIQUE KEY uq_zone_per_property (property_id, zone_name),
    INDEX idx_zone_property (property_id)
);

-- 3) Room
CREATE TABLE ROOM (
    room_id INT PRIMARY KEY AUTO_INCREMENT,
    property_id INT NOT NULL,
    room_number VARCHAR(20) NOT NULL,
    room_type VARCHAR(50),
    zone_id INT,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    CONSTRAINT fk_room_property
        FOREIGN KEY (property_id) REFERENCES PROPERTY(property_id)
        ON DELETE CASCADE ON UPDATE CASCADE,
    CONSTRAINT fk_room_zone
        FOREIGN KEY (zone_id) REFERENCES ZONE(zone_id)
        ON DELETE SET NULL ON UPDATE CASCADE,
    UNIQUE KEY uq_room_per_property (property_id, room_number),
    INDEX idx_room_zone (zone_id),
    INDEX idx_room_property (property_id)
);

-- 4) RFID tags
CREATE TABLE RFID_TAG (
    rfid_tag_id INT PRIMARY KEY AUTO_INCREMENT,
    tag_code VARCHAR(100) NOT NULL UNIQUE, -- physical card UID
    tag_type ENUM('GUEST', 'STAFF') NOT NULL,
    is_active BOOLEAN DEFAULT TRUE,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_rfid_tag_type_active (tag_type, is_active)
);

-- 5) Staff
CREATE TABLE STAFF (
    staff_id INT PRIMARY KEY AUTO_INCREMENT,
    property_id INT NOT NULL,
    name VARCHAR(100) NOT NULL,
    role VARCHAR(50),
    phone VARCHAR(20),
    status ENUM('ACTIVE', 'INACTIVE') DEFAULT 'ACTIVE',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    CONSTRAINT fk_staff_property
        FOREIGN KEY (property_id) REFERENCES PROPERTY(property_id)
        ON DELETE CASCADE ON UPDATE CASCADE,
    INDEX idx_staff_property_status (property_id, status)
);

-- 6) Guest
CREATE TABLE GUEST (
    guest_id INT PRIMARY KEY AUTO_INCREMENT,
    property_id INT NOT NULL,
    name VARCHAR(100) NOT NULL,
    phone VARCHAR(20),
    status ENUM('CHECKED_IN', 'CHECKED_OUT') DEFAULT 'CHECKED_IN',
    check_in_time DATETIME,
    check_out_time DATETIME,
    room_id INT,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    CONSTRAINT fk_guest_property
        FOREIGN KEY (property_id) REFERENCES PROPERTY(property_id)
        ON DELETE CASCADE ON UPDATE CASCADE,
    CONSTRAINT fk_guest_room
        FOREIGN KEY (room_id) REFERENCES ROOM(room_id)
        ON DELETE SET NULL ON UPDATE CASCADE,
    INDEX idx_guest_property_status (property_id, status),
    INDEX idx_guest_room (room_id)
);

-- 7) RFID assignment history (replaces staff.rfid_tag_id and guest.rfid_tag_id)
CREATE TABLE RFID_ASSIGNMENT (
    assignment_id INT PRIMARY KEY AUTO_INCREMENT,
    rfid_tag_id INT NOT NULL,
    assignee_type ENUM('GUEST', 'STAFF') NOT NULL,
    assignee_id INT NOT NULL,
    assigned_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    released_at DATETIME NULL,
    is_active BOOLEAN NOT NULL DEFAULT TRUE,
    notes VARCHAR(255),

    -- generated keys to enforce only one active assignment:
    -- one active per tag
    active_tag_key INT GENERATED ALWAYS AS (
        CASE WHEN is_active = TRUE THEN rfid_tag_id ELSE NULL END
    ) VIRTUAL,
    -- one active per assignee
    active_assignee_key VARCHAR(120) GENERATED ALWAYS AS (
        CASE WHEN is_active = TRUE THEN CONCAT(assignee_type, '#', assignee_id) ELSE NULL END
    ) VIRTUAL,

    CONSTRAINT fk_assignment_tag
        FOREIGN KEY (rfid_tag_id) REFERENCES RFID_TAG(rfid_tag_id)
        ON DELETE RESTRICT ON UPDATE CASCADE,

    UNIQUE KEY uq_active_tag (active_tag_key),
    UNIQUE KEY uq_active_assignee (active_assignee_key),
    INDEX idx_assignment_lookup (assignee_type, assignee_id, is_active),
    INDEX idx_assignment_tag_active (rfid_tag_id, is_active)
);

-- 8) RFID readers
CREATE TABLE RFID_READER (
    reader_id INT PRIMARY KEY AUTO_INCREMENT,
    property_id INT NOT NULL,
    zone_id INT NOT NULL,
    reader_connection VARCHAR(150),   -- IP/serial/identifier
    reader_name VARCHAR(100),
    status ENUM('ONLINE', 'OFFLINE', 'MAINTENANCE') DEFAULT 'ONLINE',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    CONSTRAINT fk_reader_property
        FOREIGN KEY (property_id) REFERENCES PROPERTY(property_id)
        ON DELETE CASCADE ON UPDATE CASCADE,
    CONSTRAINT fk_reader_zone
        FOREIGN KEY (zone_id) REFERENCES ZONE(zone_id)
        ON DELETE CASCADE ON UPDATE CASCADE,
    INDEX idx_reader_zone_status (zone_id, status),
    INDEX idx_reader_property (property_id)
);

-- 9) Movement logs
CREATE TABLE MOVEMENT_TAG (
    movement_id BIGINT PRIMARY KEY AUTO_INCREMENT,
    scan_time DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    rfid_tag_id INT NOT NULL,
    reader_id INT NOT NULL,
    event_type ENUM('ENTRY', 'EXIT', 'PING') DEFAULT 'PING',
    signal_strength INT NULL,
    raw_payload JSON NULL,
    CONSTRAINT fk_movement_tag
        FOREIGN KEY (rfid_tag_id) REFERENCES RFID_TAG(rfid_tag_id)
        ON DELETE RESTRICT ON UPDATE CASCADE,
    CONSTRAINT fk_movement_reader
        FOREIGN KEY (reader_id) REFERENCES RFID_READER(reader_id)
        ON DELETE RESTRICT ON UPDATE CASCADE,
    INDEX idx_movement_time (scan_time),
    INDEX idx_movement_tag_time (rfid_tag_id, scan_time),
    INDEX idx_movement_reader_time (reader_id, scan_time)
);

-- 10) Task
CREATE TABLE TASK (
    task_id INT PRIMARY KEY AUTO_INCREMENT,
    property_id INT NOT NULL,
    task_type VARCHAR(100),
    status ENUM('PENDING', 'ASSIGNED', 'COMPLETED', 'CANCELLED') DEFAULT 'PENDING',
    priority ENUM('LOW', 'MEDIUM', 'HIGH', 'CRITICAL') DEFAULT 'MEDIUM',
    created_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    assigned_time DATETIME,
    complete_time DATETIME,
    zone_id INT,
    room_id INT,
    staff_id INT,
    description VARCHAR(255),
    CONSTRAINT fk_task_property
        FOREIGN KEY (property_id) REFERENCES PROPERTY(property_id)
        ON DELETE CASCADE ON UPDATE CASCADE,
    CONSTRAINT fk_task_zone
        FOREIGN KEY (zone_id) REFERENCES ZONE(zone_id)
        ON DELETE SET NULL ON UPDATE CASCADE,
    CONSTRAINT fk_task_room
        FOREIGN KEY (room_id) REFERENCES ROOM(room_id)
        ON DELETE SET NULL ON UPDATE CASCADE,
    CONSTRAINT fk_task_staff
        FOREIGN KEY (staff_id) REFERENCES STAFF(staff_id)
        ON DELETE SET NULL ON UPDATE CASCADE,
    INDEX idx_task_status_priority (status, priority),
    INDEX idx_task_zone (zone_id),
    INDEX idx_task_staff (staff_id),
    INDEX idx_task_property (property_id)
);

-- 11) Staff-to-zone force allocation
CREATE TABLE STAFF_ZONE_ALLOCATION (
    allocation_id INT PRIMARY KEY AUTO_INCREMENT,
    property_id INT NOT NULL,
    staff_id INT NOT NULL,
    zone_id INT NOT NULL,
    allocated_by_staff_id INT NULL, -- admin/supervisor from STAFF
    priority ENUM('LOW', 'MEDIUM', 'HIGH', 'CRITICAL') DEFAULT 'MEDIUM',
    status ENUM('PLANNED', 'ACTIVE', 'ENDED', 'CANCELLED') DEFAULT 'PLANNED',
    start_time DATETIME NOT NULL,
    end_time DATETIME NULL,
    reason VARCHAR(255),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    CONSTRAINT fk_alloc_property
        FOREIGN KEY (property_id) REFERENCES PROPERTY(property_id)
        ON DELETE CASCADE ON UPDATE CASCADE,
    CONSTRAINT fk_alloc_staff
        FOREIGN KEY (staff_id) REFERENCES STAFF(staff_id)
        ON DELETE CASCADE ON UPDATE CASCADE,
    CONSTRAINT fk_alloc_zone
        FOREIGN KEY (zone_id) REFERENCES ZONE(zone_id)
        ON DELETE CASCADE ON UPDATE CASCADE,
    CONSTRAINT fk_alloc_by_staff
        FOREIGN KEY (allocated_by_staff_id) REFERENCES STAFF(staff_id)
        ON DELETE SET NULL ON UPDATE CASCADE,
    INDEX idx_alloc_zone_status_time (zone_id, status, start_time),
    INDEX idx_alloc_staff_status (staff_id, status),
    INDEX idx_alloc_property (property_id)
);
